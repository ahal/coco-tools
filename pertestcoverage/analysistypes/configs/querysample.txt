select distinct jn.text as fixed_by_revision, jt.name as job_type_name, repository.name as repo, SUBSTRING_INDEX(fl.test, ' ', 1) as testname from repository, job_note jn, job j, failure_line as fl, push p, job_type jt, reference_data_signatures sig, option_collection oc, `option` op, build_platform bp where jn.failure_classification_id=2 and jn.text!='' and jn.created>'2018-08-28' and jn.job_id=j.id and j.job_type_id=jt.id and j.signature_id=sig.id and sig.option_collection_hash=oc.option_collection_hash and oc.option_id=op.id and jt.name like 'test-%' and j.build_platform_id=bp.id and j.repository_id=repository.id and (repository.name='mozilla-inbound' or repository.name='autoland') and j.push_id=p.id and fl.job_guid=j.guid and fl.test!='' and (job_type_name like '%xpcshell%' or job_type_name like '%mochitest%') and job_type_name like '%linux64/%' and fl.test not like 'Main app process exited normally' and fl.test not like 'Last test finished' and fl.test not like 'automation.py' group by SUBSTRING(fixed_by_revision, 1, 12);
